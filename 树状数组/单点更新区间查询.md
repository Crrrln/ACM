## 单点更新区间查询

### 代码
```cpp

const int maxn=1e5+10;
int n;
int a[maxn],c[maxn]; //对应原数组和树状数组

int lowbit(int x){
    return x&(-x);
}

//在i位置加上k
void update(int i,int k){
    while(i <= n){
        c[i] += k;
        i += lowbit(i);
    }
}

//求A[1 - i]的和
int getsum(int i){
    int res = 0;
    while(i > 0){
        res += c[i];
        i -= lowbit(i);
    }
    return res;
}
```
